STEP 1: Create Rails API App
rails new currency_api --api
cd currency_api
 STEP 2: Generate Scaffold
rails generate scaffold Conversion amount:decimal from_currency:string to_currency:string converted_amount:decimal
 STEP 3: Run Migration
rails db:migrate
STEP 4: Add Custom Route (/convert)

Open: config/routes.rb

Rails.application.routes.draw do
  resources :conversions do
    collection do
      post 'convert'
    end
  end
end
 STEP 5: Add Logic in Controller

Open: app/controllers/conversions_controller.rb

Add this method:

def convert
  amount = params[:amount].to_f
  from = params[:from_currency]
  to = params[:to_currency]

  # Static exchange rates
  rates = {
    "USD" => 1.0,
    "EUR" => 0.9,
    "INR" => 83.0
  }

  if rates[from] && rates[to]
    converted_amount = (amount / rates[from]) * rates[to]

    render json: {
      amount: amount,
      from: from,
      to: to,
      converted_amount: converted_amount.round(2)
    }
  else
    render json: { error: "Invalid currency" }, status: 422
  end
end
